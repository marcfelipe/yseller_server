Future<List> listar_chamados(String url_, String encodedPath) async{
  //print("iniciando listar_chamados()");
  var headers_req =<String, String>{'Content-Type': "application/json"};
  var uri = Uri.http(url_, encodedPath);
  var response = await http.get(uri, headers: headers_req);
  print(response.body.runtimeType);
  //print(response.body);
  return json.decode(response.body);
}
